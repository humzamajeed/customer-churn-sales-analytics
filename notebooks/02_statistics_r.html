3. Hypothesis Test 1: T-Test — Tenure by Churn

H₀: Mean tenure is the same for churned and non-churned customers
H₁: Churned customers have significantly lower tenure

# Run the t-test
t_result <- df %>%
  t_test(tenure ~ Churn, alternative = "two.sided") %>%
  add_significance()

print(t_result)
## # A tibble: 1 × 7
##   statistic  t_df   p_value alternative estimate lower_ci upper_ci
##       <dbl> <dbl>     <dbl> <chr>          <dbl>    <dbl>    <dbl>
## 1      35.0 4046. 2.35e-234 two.sided       19.7     18.6     20.8
# Visualize
ggboxplot(df, x = "Churn", y = "tenure",
          color = "Churn", palette = c("steelblue", "salmon"),
          add = "mean_ci",
          title = "Tenure by Churn Status",
          xlab = "Churn", ylab = "Tenure (months)") +
  stat_compare_means(method = "t.test", label.y = 75)

Interpretation: If p-value < 0.05, we reject H₀ — churned customers have significantly different tenure.

## 5. Hypothesis Test 3: Chi-Square — Churn vs Contract Type

H₀: Churn is independent of Contract type
H₁: There is a significant association between Churn and Contract type

# Contingency table
contract_table <- table(df$Contract, df$Churn)
print(contract_table)
##                 
##                    No  Yes
##   Month-to-month 2220 1655
##   One year       1306  166
##   Two year       1637   48
# Chi-square test
chisq_result <- chisq.test(contract_table)
print(chisq_result)
## 
##  Pearson's Chi-squared test
## 
## data:  contract_table
## X-squared = 1179.5, df = 2, p-value < 2.2e-16
# Visualize
df %>%
  group_by(Contract, Churn) %>%
  summarise(Count = n(), .groups = "drop") %>%
  group_by(Contract) %>%
  mutate(Percentage = Count / sum(Count) * 100) %>%
  filter(Churn == "Yes") %>%
  ggplot(aes(x = Contract, y = Percentage, fill = Contract)) +
  geom_col(show.legend = FALSE) +
  scale_fill_manual(values = c("coral", "steelblue", "seagreen")) +
  labs(title = "Churn Rate by Contract Type",
       x = "Contract Type", y = "Churn Rate (%)") +
  theme_minimal()

Interpretation: If p-value < 0.05, contract type significantly affects whether a customer churns.

7. Confidence Intervals — Monthly Charges by Churn Group

ci_results <- df %>%
  group_by(Churn) %>%
  summarise(
    Mean  = mean(MonthlyCharges),
    SE    = sd(MonthlyCharges) / sqrt(n()),
    Lower = Mean - 1.96 * SE,
    Upper = Mean + 1.96 * SE
  )

print(ci_results)

8. Statistical Test Summary

# T-test for Tenure
t_result <- t.test(tenure ~ Churn, data = df)

# T-test for Monthly Charges
t_result2 <- t.test(MonthlyCharges ~ Churn, data = df)

# Chi-Square tests
chisq_result <- chisq.test(table(df$Churn, df$Contract))
chisq_payment <- chisq.test(table(df$Churn, df$PaymentMethod))
## 8. Statistical Test Summary
# Create the data frame with corrected p-value references
summary_df <- data.frame(
  Test = c(
    "T-Test: Tenure vs Churn",
    "T-Test: MonthlyCharges vs Churn",
    "Chi-Square: Churn vs Contract",
    "Chi-Square: Churn vs PaymentMethod"
  ),
  Method = c("Independent t-test", "Independent t-test",
             "Chi-Square", "Chi-Square"),
  P_Value = c(
    format.pval(t_result$p.value, eps = .0001, digits = 4),
    format.pval(t_result2$p.value, eps = .0001, digits = 4),
    format.pval(chisq_result$p.value, eps = .0001, digits = 4),
    format.pval(chisq_payment$p.value, eps = .0001, digits = 4)
  ),
  Significant = ifelse(
    c(t_result$p.value, t_result2$p.value, 
      chisq_result$p.value, chisq_payment$p.value) < 0.05,
    "Yes ✅", "No ❌"
  )
)
